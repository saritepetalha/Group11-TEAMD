rectangle "Render" as Render

' Message flow
Actor - Game: run()

Playing -down- TileManager :  ↓ 1.1: getSprite()

Playing -down- EditTiles : ↓ 1.2: draw()

EditTiles -down- EditTiles : 1.2.1: drawButtons()

EditTiles -down- ButtonLoop : ↓ 1.2.2*[i=1..n]: draw()

EditTiles -left- TileManager : → 1.2.3: getSprite()

Playing -down- Playing : 1.3: drawSelectedTile()
