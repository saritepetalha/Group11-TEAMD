== Start Flow ==
"System Timer" -> System: triggerWaveStart()

alt Game is not paused
    System -> System: spawnEnemyGroup()
    loop While group not complete
        System -> System: wait(spawnInterval)
        System -> System: spawnNextEnemy()
    end

    loop Between groups
        System -> System: wait(groupDelay)
        System -> System: spawnEnemyGroup()
    end

    System -> System: wait(waveCooldown)

    note right of System
        Wave complete. Next wave will be scheduled.
    end note

else Game is paused
    System -> System: delayWaveStart()
    note right of System
        Spawning postponed until unpaused.
    end note
end
