== Main Scenario ==
Player -> System: clickSaveButton()

System -> System: validateSettings()

alt Settings are valid
    System -> FileSystem: saveSettingsToDisk()
    
    alt Write successful
        FileSystem --> System: success
        System -> Player: showConfirmation("Settings saved")
        note right of System
            Settings retained for future sessions.
        end note

    else Write fails
        FileSystem --> System: error
        System -> Player: showError("Failed to save. Using previous settings.")
        System -> System: keepPreviousSettings()
    end

else Settings are invalid
    System -> Player: showError("Invalid settings. Cannot save.")
end
